  file(MAKE_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/comb)
  file(MAKE_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/partition)

  cxx_gtest(TestModuleSortGenLogs "${MODULE_SORT_GEN_LOGS_SRCS}" ${SHA_SRCS})
